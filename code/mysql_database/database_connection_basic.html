Slq Database
    class DBConnectionService {
        private $host = "localhost";
        private $userName = "root";
        private $password = "";
        private $db = "db_name";
        private $con;
        
        function __contruct() {
            $this->con = mysqli_connect($this->host, $this->userName, $this->password, $this->db);
        }
    
        function getConnection () {
            if(!$this->con) {
                return NULL;
            } else {
                return $this->con;
            }
        }
    }
